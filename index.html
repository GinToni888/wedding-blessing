    <div class="card">

        <h2>送上祝福</h2>

        <input type="text" id="name" placeholder="你的大名" required>

        <textarea id="message" rows="4" placeholder="想對新人說的話..." required></textarea>

        <button onclick="sendBlessing()">送出祝福</button>

    </div>

 

    <script>

        // 初始化 LIFF

        liff.init({ liffId: "2008922628-n0ex971B" }).then(() => {

            if (!liff.isLoggedIn()) { liff.login(); }

        });

 

        async function sendBlessing() {

            const name = document.getElementById('name').value;

            const message = document.getElementById('message').value;

            const gasUrl = "https://script.google.com/macros/s/AKfycby-P94fg0kooAipQFNCgBejG2EFx0JDT-8gC0A6mlEwISuEyUpyqBJk6XTrAR7FXGcWvQ/exec"; // 貼上你的 GAS URL

 

            if (!name || !message) return alert("請填寫姓名與祝福語");

 

            const response = await fetch(gasUrl, {

                method: "POST",

                body: JSON.stringify({ name, message })

            });

 

            if (response.ok) {

                alert("祝福已送到！謝謝你");

                liff.closeWindow(); // 成功後自動關閉視窗

            } else {

                alert("傳送失敗，請稍後再試");

            }

        }

    </script>
